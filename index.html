        // Countdown to March 20, 2026

        const releaseDate = new Date("March 20, 2026 00:00:00").getTime();

        function updateTimer() {

            const now = new Date().getTime();

            const dist = releaseDate - now;

            const d = Math.floor(dist / (1000 * 60 * 60 * 24));

            const h = Math.floor((dist % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));

            const m = Math.floor((dist % (1000 * 60 * 60)) / (1000 * 60));

            const s = Math.floor((dist % (1000 * 60)) / 1000);


            document.getElementById("days").innerText = d.toString().padStart(2, '0');

            document.getElementById("hours").innerText = h.toString().padStart(2, '0');

            document.getElementById("minutes").innerText = m.toString().padStart(2, '0');

            document.getElementById("seconds").innerText = s.toString().padStart(2, '0');


            if (dist < 0) {

                document.getElementById("timer").innerHTML = "<div class='syne text-3xl text-teal-400 uppercase tracking-tighter animate-pulse italic'>The Era Has Begun</div>";

            }

        }

        setInterval(updateTimer, 1000);

        updateTimer();
